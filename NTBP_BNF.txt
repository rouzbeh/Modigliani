neuron      ::= NEURON '{' membrane '}'
membrane    ::= membrane_compartment ( ion_channel* )

membrane_compartment ::== ( SPHERE | CYLINDER ) membrane_compartment_param
membrane_compartment_param ::= membrane_resitance | membrane_capacitance | radius


ion_channel       ::= ION_CHANNEL ion_channel_model '{' ( ion_channel_param* ) '}'
ion_channel_param ::= ion_channel_algorithm | single_channel_conductance | q10 | baseTemperature

ion_channel_model          ::= NTBP_NA_SGA_HH | NTBP_NA_PATLAK | NTBP_NA_COLBERT_AXONAL | NTBP_NA_HRANVIER
ion_channel_algorithm      ::= NTBP_DETERMINISTIC | NTBP_LANGEVIN | NTBP_SINGLE_CHANNEL | NTBP_BIONOMIALPOPULATION | NTBP_GILLESPIE | NTBP_NOISY_MEAN
single_channel_conductance ::= REAL
                       q10 ::= REAL
	   baseTemperature ::= REAL
	   
membrane_compartment ::=  spherical_membrane_compartment | cylindrical_membrane_compartment
